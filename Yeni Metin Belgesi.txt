using System;
using System.Windows.Input;

namespace MustafaBayram.Commands
{
    public class RelayCommand : ICommand
    {
        private Action _execute1;
        private Func<bool> _canExecute1;
        private Action<object> _execute2;
        private Func<object, bool> _canExecute2;

        public RelayCommand(Action<object> execute) : this(execute, null) { }

        public RelayCommand(Action<object> execute, Func<object, bool> canExecute)
        {
            this._execute2 = execute;
            this._canExecute2 = canExecute;
        }

        public RelayCommand(Action execute) : this(execute, null) { }

        public RelayCommand(Action execute, Func<bool> canExecute)
        {
            this._execute1 = execute;
            this._canExecute1 = canExecute;
        }

        public bool CanExecute(object parameter)
        {
            if (_canExecute1 != null)
                return _canExecute1.Invoke();
            else if (_canExecute2 != null)
                return _canExecute2.Invoke(parameter);
            else
                return true;
        }

        public void Execute(object parameter)
        {
            if (_execute1 != null)
                _execute1.Invoke();
            else if (_execute2 != null)
                _execute2.Invoke(parameter);
        }

        public event EventHandler CanExecuteChanged
        {
            add { CommandManager.RequerySuggested += value; }
            remove { CommandManager.RequerySuggested -= value; }
        }
    }
}

using System;
using System.ComponentModel;
using System.Runtime.CompilerServices;

namespace MustafaBayram.ViewModels.Base
{
    public class ObservableObject : INotifyPropertyChanged
    {
        protected virtual void SetValue<T>(ref T variable, T value, [CallerMemberName] string propertyName = null)
        {
            if (Object.Equals(variable, value)) return;
            variable = value;
            OnPropertyChanged(propertyName);
        }

        public event PropertyChangedEventHandler PropertyChanged;
        protected virtual void OnPropertyChanged([CallerMemberName] string propertyName = null)
        {
            PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(propertyName));
        }
    }
}

using MustafaBayram.Commands;
using MustafaBayram.ViewModels.Base;
using System.Windows;
using System.Windows.Controls;
using System.Linq;
using MustafaBayram.BusinessLayer;
using MustafaBayram.EntityLayer.Models;
using MustafaBayram.Models;
using MustafaBayram.Views.LoginViews;

namespace MustafaBayram.ViewModels.LoginWievModels
{
    public class LoginViewModel : ObservableObject
    {
        private string eposta;
        public string EPosta { get => eposta; set => SetValue(ref eposta, value); }

        public RelayCommand LoginCommand { get; set; }

        public LoginViewModel()
        {
            LoginCommand = new RelayCommand(p => {
                var passwordBox = p as PasswordBox;
                if (passwordBox == null) return;

                string sifre = passwordBox.Password;

                var kullanici = new KullaniciManager().ToList()
                    .FirstOrDefault(x => x.Eposta == EPosta && x.Parola == sifre);

                if (kullanici != null)
                {
                    LoginVM.Id = kullanici.Id;
                    LoginVM.EPosta = kullanici.Eposta;
                    LoginVM.Parola = kullanici.Parola;

                    MainWindow anaPencere = new MainWindow();
                    anaPencere.Show();

                    foreach (Window item in Application.Current.Windows)
                    {
                        if (item is LoginWindow) { item.Close(); break; }
                    }
                }
                else
                {
                    MessageBox.Show("E-Posta veya şifre hatalı!", "Hata", MessageBoxButton.OK, MessageBoxImage.Error);
                }
            });
        }
    }
}

using MustafaBayram.Commands;
using MustafaBayram.EntityLayer.Models;
using MustafaBayram.ViewModels.Base;
using System.Windows;

namespace MustafaBayram.ViewModels.KullaniciWievModels
{
    public class KullaniciEditViewModel : ObservableObject
    {
        private Kullanici model;
        public Kullanici Model { get => model; set => SetValue(ref model, value); }

        public RelayCommand TamamCommand { get; set; }
        public RelayCommand IptalCommand { get; set; }

        public KullaniciEditViewModel() : this(new Kullanici()) { }

        public KullaniciEditViewModel(Kullanici model)
        {
            this.Model = model;

            TamamCommand = new RelayCommand(OnTamam);
            IptalCommand = new RelayCommand(OnIptal);
        }

        private void OnTamam(object parameter)
        {
            if (parameter is Window view)
            {
                view.DialogResult = true;
            }
        }

        private void OnIptal(object parameter)
        {
            if (parameter is Window view)
            {
                view.DialogResult = false;
            }
        }
    }
}

using MustafaBayram.BusinessLayer;
using MustafaBayram.Commands;
using MustafaBayram.EntityLayer.Models;
using MustafaBayram.ViewModels.Base;
using MustafaBayram.Views.KullaniciViews;
using System.Collections.ObjectModel;
using System.Windows;

namespace MustafaBayram.ViewModels.KullaniciWievModels
{
    public class KullaniciListViewModel : BaseListViewModel<Kullanici>
    {
        private readonly KullaniciManager manager;

        public RelayCommand RefreshCommand { get; set; }
        public RelayCommand InsertCommand { get; set; }
        public RelayCommand DeleteCommand { get; set; }
        public RelayCommand UpdateCommand { get; set; }

        public KullaniciListViewModel()
        {
            this.manager = new KullaniciManager();

            RefreshCommand = new RelayCommand(OnRefresh);
            InsertCommand = new RelayCommand(OnInsert);
            DeleteCommand = new RelayCommand(OnDelete, (param) => param != null);
            UpdateCommand = new RelayCommand(OnUpdate, (param) => param != null);

            OnRefresh();
        }

        private void OnRefresh()
        {
            var kullanicilar = manager.ToList();
            Items = new ObservableCollection<Kullanici>(kullanicilar);
        }

        private void OnInsert()
        {
            var vm = new KullaniciEditViewModel();
            var view = new KullaniciWindow() { DataContext = vm };

            if (view.ShowDialog() == true)
            {
                var entity = manager.Add(new Kullanici
                {
                    Eposta = vm.Model.Eposta,
                    Parola = vm.Model.Parola
                });

                vm.Model.Id = entity.Id;
                Items.Add(vm.Model);

                OnRefresh();
            }
        }

        private void OnDelete(object parameter)
        {
            if (parameter is Kullanici kullanici)
            {
                string mesaj = $"{kullanici.Eposta} kullanıcısını silmek istiyor musunuz?";

                if (MessageBoxResult.Yes == MessageBox.Show(mesaj, "Kullanıcı Sil", MessageBoxButton.YesNo, MessageBoxImage.Question))
                {
                    if (manager.Delete(kullanici.Id) > 0)
                    {
                        Items.Remove(kullanici);
                    }
                }
            }
        }

        private void OnUpdate(object parameter)
        {
            if (parameter is Kullanici kullanici)
            {
                var vm = new KullaniciEditViewModel(kullanici);
                var view = new KullaniciWindow() { DataContext = vm };

                if (view.ShowDialog() == true)
                {
                    if (0 < manager.Update(new Kullanici
                    {
                        Id = vm.Model.Id,
                        Eposta = vm.Model.Eposta,
                        Parola = vm.Model.Parola
                    }))
                    {
                        OnRefresh();
                    }
                }
            }
        }
    }
}
